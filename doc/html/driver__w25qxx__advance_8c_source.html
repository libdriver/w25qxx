51uint8_t w25qxx_advance_init(w25qxx_type_t type, w25qxx_interface_t interface, w25qxx_bool_t dual_quad_spi_enable)
52{
53 uint8_t res;
54
55 /* link interface function */
56 DRIVER_W25QXX_LINK_INIT(&gs_handle, w25qxx_handle_t);
57 DRIVER_W25QXX_LINK_SPI_QSPI_INIT(&gs_handle, w25qxx_interface_spi_qspi_init);
58 DRIVER_W25QXX_LINK_SPI_QSPI_DEINIT(&gs_handle, w25qxx_interface_spi_qspi_deinit);
59 DRIVER_W25QXX_LINK_SPI_QSPI_WRITE_READ(&gs_handle, w25qxx_interface_spi_qspi_write_read);
60 DRIVER_W25QXX_LINK_DELAY_MS(&gs_handle, w25qxx_interface_delay_ms);
61 DRIVER_W25QXX_LINK_DELAY_US(&gs_handle, w25qxx_interface_delay_us);
62 DRIVER_W25QXX_LINK_DEBUG_PRINT(&gs_handle, w25qxx_interface_debug_print);
63
64 /* set chip type */
65 res = w25qxx_set_type(&gs_handle, type);
66 if (res != 0)

68 w25qxx_interface_debug_print("w25qxx: set type failed.\n");
69
70 return 1;
71 }
72

75 if (res != 0)
76 {
77 w25qxx_interface_debug_print("w25qxx: set interface failed.\n");
78
79 return 1;
80 }
81
82 /* set dual quad spi */
83 res = w25qxx_set_dual_quad_spi(&gs_handle, dual_quad_spi_enable);
84 if (res != 0)
85 {
86 w25qxx_interface_debug_print("w25qxx: set dual quad spi failed.\n");

90 }
91
92 /* chip init */
93 res = w25qxx_init(&gs_handle);
94 if (res != 0)
95 {
96 w25qxx_interface_debug_print("w25qxx: init failed.\n");
97
98 return 1;
99 }
100 else
101 {
102 if (type >= W25Q256)
103 {
104 res = w25qxx_set_address_mode(&gs_handle, W25QXX_ADDRESS_MODE_4_BYTE);
105 if (res != 0)
106 {
107 w25qxx_interface_debug_print("w25qxx: set address mode failed.\n");
108 (void)w25qxx_deinit(&gs_handle);
109
110 return 1;
111 }
112 }
113
114 return 0;
115 }
